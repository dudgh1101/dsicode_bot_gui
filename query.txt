-- 유저 테이블
create table users(
	user_id varchar(30), -- use유저 아이디(로그인용)
	user_email varchar(50), -- 유저 이메일(본인인증용)
	password varchar(100) not null, -- 비밀번호
	nickname varchar(20) not null, -- 유저 닉네임(보이는 이름)
	phone_number varchar(20), -- 유저 전화번호
	u_status char(5) not null, -- 유저 상태(온라인, 오프라인,자리비움등등)
	no_edit boolean default false, -- 상태에 대한 변경 허용상태
	create_at time default NOW(), --언제 생성된 유저인지
	will_read boolean default false, -- 읽을 메시지가 존재하는지
	login_count int default 3, --로그인 제한 횟수(기본 3회)
	constraint fk_users_status FOREIGN KEY (u_status) REFERENCES status_rule(status_type),
	constraint pk_user_id primary key(user_id)
	)ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 유저 삭제 및 수정되기 이전 원본 데이터 테이블
create table user_gavige(
	user_gavige_id varchar(10), -- 원본데이터 고유 코드(아래는 유저 테이블과 같음)
	user_id varchar(30),
	user_email varchar(50),
	password varchar(100) not null,
	nickname varchar(20) not null,
	phone_number varchar(20),
	u_status char(5) not null, -- 상태(삭제, 수정 등등)
	no_edit boolean default false,
	create_at time default NOW(),
	will_read boolean default false,
	login_count int default 3,
	constraint pk_user_gavige_id primary key (user_gavige_id),
	constraint fk_users_status_gavige FOREIGN KEY (u_status) REFERENCES status_rule(status_type)
	)ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

create table user_info_log(

	act_type char(5), -- 행동 코드
	act_day time default now(), -- 언제 했는지
	user_id varchar(30), -- 누가했는지
	result char(5), -- 결과 코드

)ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
